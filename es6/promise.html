JavaScript Promise

Promise 是构造函数，用来构造实例对象，在创建实例对象的时候，将一个函数作为参数传递给 Promise，该函数会在执行 Promise 构造函数时执行，并且 Promise 会传递两个实参函数（resolve 和 reject）给这个函数执行。

传递的函数内部可编写异步代码，当异步任务完成时，通过调用 resolve 或 reject 函数来通知 Promise 实例对象，让他修改表示状态的自身属性。

一旦异步代码执行完，并且执行了 resolve 或 reject 函数，这时候该 Promise 实例对象的状态就冻结了。

当程序员需要用到异步任务后得到的数据时，只需要找到 Promise 实例对象，调用它的实例方法 then，最多传入两个函数作为接收数据的回调函数。

第一个函数表示成功的回调，也就是当异步任务完成时（程序员自己判断是否成功，在构造时传递的函数中写的代码，如果成功调用 resolve 函数，如果失败，调用 reject 函数），Promise 实例对象会自己执行 then 的第一个参数函数。

执行的时间根据异步任务的完成情况和通过 then 传递的回调函数的具体写在哪里来执行。

如果程序员写好了 then 的回调，而异步任务还没结束，那这个异步任务会进入当前 Promise 实例对象的待执行成功回调函数所组成的数组中，等待 resolve 函数执行改变 Promise 状态并执行数组中的回调函数。

如果程序员写好了 then 的回调，而异步任务早就结束，这时候会直接执行回调函数。

同样道理，then 的第二个参数就是错误回调，当程序员认为异步任务出错了，则执行 reject 函数，这时候 Promise 实例对象的状态就从 pending 变为 rejected，也会执行错误回调函数。

        const p = new Promise((resolve) => {
            setTimeout(() => {
                resolve();
                console.log("2. 异步任务执行成功")
            }, 5000);
            console.log('1. 注册异步任务成功')
        })
        // 实例化 Promise 对象后立即注册成功回调，这时候异步任务还没结束，等异步任务结束，Promise 实例对象状态改变，才会执行成功回调
        p.then(() => {
            console.log("3. 正在执行异步任务执行成功的回调函数")
        })

为什么 Promise 构造函数中定义的 resolve 和 reject 保存在一个数组中，因为一个 Promise 实例对象可能执行多个 then 方法从而传递多个回调函数 


        const p = new Promise(resolve => {
            setTimeout(() => {
                resolve(2);
            }, 3000);
        })

        // 传递回调函数的时候异步任务还没结束，也就意味着 resolve 还没执行
        p.then(res => {
            console.log('1. get res', res)
        })
        p.then(res => {
            console.log('2. get res', res)
        })

自定义 Promise 是前段必由之路

    function MyPromise(executor) {

            const self = this;
            this.status = 'pending'
            this.resolvedCallbackList = []
            this.rejectedCallbackList = []

            function resolve(value) {
                console.log('run resolve', value)
                if (self.status === 'pending') {
                    self.status = 'resolved'
                    console.log('run resolve', value, self.resolvedCallbackList)
                    for (let i = 0; i < self.resolvedCallbackList.length; i++) {
                        self.resolvedCallbackList[i](value)
                    }
                }
            }

            function reject(reason) {
                if (this.status === 'pending') {
                    this.status = 'rejected'
                    for (let i = 0; i < this.rejectedCallbackList.length; i++) {
                        this.rejectedCallbackList[i](reason)
                    }
                }
            }

            try {
                // 运行程序员自己写的代码（主要是异步代码），程序员自己判断异步的结果是否成功，如果正确执行 MyPromise 给它传入的 resolve 函数，如果失败执行 reject 函数
                executor(resolve, reject)
            } catch (error) {
                // executor  发生了同步错误，会在这里捕获
                reject(error)
            }

        }

        MyPromise.prototype.then = function (onResolved, onRejected) {
            // if (this.status === 'resolved') {

            // }
            if (this.status === 'pending') {
                this.resolvedCallbackList.push(onResolved)
            }
        }

        const p = new MyPromise((resolve, reject) => {
            setTimeout(() => {
                resolve(3)
            }, 3000);
        })

        console.log(p)

        p.then(res => {
            console.log('res', res)
        })

一步一步实现 Promise，还差链式调用和错误处理

            function MyPromise(executor) {
        const self = this;
        self.status = 'pending'
        self.resolvedCallbackList = []
        self.rejectedCallbackList = []
        self.data = undefined;
        self.error = undefined;

        function resolve(value) {
            setTimeout(() => {
                if (self.status === 'pending') {
                    self.status = 'resolved'
                    self.data = value; // 如果 then 注册的回调函数在 promise status 已经成功的情况下才注册成功，那么只能在 then 里面判断状态立即执行回调了，
                    // 所以这里需要保存 promise 实例对象的数据

                    // 这里得用异步的写法，如果不用，会出现包裹非异步任务的 promise，比后面的同步任务还先执行， 这样就背离了 promise 的设计原则
                    setTimeout(() => {
                        console.log("run my resolve", value)
                        for (let i = 0; i < self.resolvedCallbackList.length; i++) {
                            self.resolvedCallbackList[i](value) // 如果异步任务还没执行完，成功的回调函数就通过 then 注册了，不论注册几个都会放进数组，在这里执行

                            // 执行回调函数其实本质上对于 promise 毫无影响，如果不考虑其他的 Promise 依赖当前这个 Promise 的结果来执行，那就把回调函数当作副作用函数执行即可。
                        }
                    }, 0);
                    // setTimeout 默认会被添加 4ms
                } else {
                    console.log('重复执行 resolve 是没用的')
                }
            }, 0);
        }

        function reject(error) {
            setTimeout(() => {
                if (self.status === 'pending') {
                    self.status = 'rejected'
                    self.error = error;
                    setTimeout(() => {
                        console.log('error', error)
                        for (let i = 0; i < self.rejectedCallbackList.length; i++) {
                            self.rejectedCallbackList[i](error) // 如果异步任务还没执行完，失败的回调函数就通过 then 注册了，不论注册几个都会放进数组，在这里执行
                        }
                    }, 0);
                } else {
                    console.log('用户乱写传入给 Promise 构造函数的包裹异步任务的函数，还试图将非 pending 的 promise 修改状态')
                }
            }, 0);
        }

        try {
            executor(resolve, reject)
        } catch (error) {
            // 这里捕捉的是传入的函数执行过程中发生的同步错误，而异步任务的错误由程序员自己判断指定
            reject(error)
        }
    }

        // 但 Promise 的作用还有链式调用，即通过上一个异步任务的结果来开启下一次异步任务，或者在下一次异步任务中使用当前任务的结果和之前任务的结果，一起做处理。
        // 链式的意义就是解决回调地狱，也就是回调函数的嵌套式写法
        // 把嵌套的写法转换成链式写法，最起码看起来舒服
        console.log('then status', this.status)
        if (this.status === 'resolved') {

            return new MyPromise((resolve, reject) => {
                const res = onResolved(this.data); // 回调函数处理完数据后把
                resolve(res)
            })

        }
        if (this.status === 'rejected') {
            onRejected(this.error)
        }
        if (this.status === 'pending') {
            this.resolvedCallbackList.push(onResolved)
            this.rejectedCallbackList.push(onRejected)
        }
    }
